## recherce de mots composé

{
  "query": {
    "match_phrase": {
        "answer.reasons": "not guarantee"
    }
  }

GET /java8-ocp2/_doc/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "match_phrase": {
            "question.subject": "class Diagram"
          }
        },
        {
          "match_phrase": {
            "answer.reasons": "objects"
          }
        }
      ]
    }
  }
}

GET /java8-ocp2/_doc/_search
{
  "query": {
    "bool": {
      "must_not": [
        {
          "match_phrase": {
            "question.subject": "class Diagram"
          }
        }/_/
      ]
    }
  }
}
